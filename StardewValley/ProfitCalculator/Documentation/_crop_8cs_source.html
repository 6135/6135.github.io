259 public int TotalAvailableDays(Season currentSeason, int day)
260 {
261 int totalAvailableDays = 0;
262 if (IsAvailableForCurrentSeason(currentSeason))
263 {
264 //Each season has 28 days,
265 //get index of current season
266 int seasonIndex = Array.IndexOf(Seasons.ToArray(), currentSeason);
267 //iterate over the array and add the number of days for each season that is later than the current season
268 for (int i = seasonIndex + 1; i < Seasons.Length; i++)
269 {
270 totalAvailableDays += 28;
271 }
272 //add the number of days in the current season
273 totalAvailableDays += TotalAvailableDaysInCurrentSeason(day);
274 }
275 if (currentSeason == Season.Greenhouse)
276 {
277 totalAvailableDays = (28 * 4);
278 }
279 return totalAvailableDays;

299 public int TotalHarvestsWithRemainingDays(Season currentSeason, FertilizerQuality fertilizerQuality, int day)
300 {
301 int totalHarvestTimes = 0;
302 int totalAvailableDays = TotalAvailableDays(currentSeason, day);
303
304 //season is Greenhouse
305 int growingDays = Days * (int)GetAverageGrowthSpeedValueForCrop(fertilizerQuality);
306 if (IsAvailableForCurrentSeason(currentSeason) || currentSeason == Season.Greenhouse)
307 {
308 if (totalAvailableDays < growingDays)
309 return 0;
310 //if the crop regrows, then the total harvest times are 1 for the first harvest and then the number of times it can regrow in the remaining days. We always need to subtract one to account for the day lost in the planting day.
311 if (Regrow > 0)
312 {
313 totalHarvestTimes = ((int)(1 + ((totalAvailableDays - Days) / (double)Regrow)));
314 }
315 else
316 totalHarvestTimes = totalAvailableDays / growingDays;
317 }
318 return totalHarvestTimes;
319 }
