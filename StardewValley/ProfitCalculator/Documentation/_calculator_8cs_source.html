172 private double totalCropProfit(Crop crop)
173 {
174 double totalProfitFromFirstProduce;
175 double totalProfitFromRemainingProduce;
176
177 if (!crop.affectByQuality)
178 {
179 totalProfitFromFirstProduce = 0;
180 totalProfitFromRemainingProduce = crop.Price;
181 }
182 else
183 {
184 double averageValue = crop.Price * this.getAverageValueForCropAfterModifiers();//only applies to first produce
185 totalProfitFromFirstProduce = averageValue;
186
187 double averageExtraCrops = crop.AverageExtraCropsFromRandomness();
188
189 totalProfitFromRemainingProduce = (crop.MinHarvests - 1 >= 0 ? crop.MinHarvests - 1 : 0) * crop.Price;
190
191 totalProfitFromRemainingProduce += (crop.Price * averageExtraCrops);
192 }
193 if (!UseBaseStats && Game1.player.professions.Contains(Farmer.tiller))
194 {
195 totalProfitFromRemainingProduce *= 1.1f;
196 }
197 double result = (totalProfitFromFirstProduce + totalProfitFromRemainingProduce) * crop.TotalHarvestsWithRemainingDays(Season, FertilizerQuality, (int)Day);
198 return result;
199 }

314 {

341 private double getCropBaseGoldQualityChance(double limit = 9999999999)
342 {
343 int fertilizerQualityLevel = ((int)FertilizerQuality) > 0 ? ((int)FertilizerQuality) : 0;
344 double part1 = (0.2 * (farmingLevel / 10.0)) + 0.01;
345 double part2 = 0.2 * (fertilizerQualityLevel * ((farmingLevel + 2) / 12.0));
346 return Math.Min(limit, part1 + part2);

349 private double getCropBaseQualityChance()
350 {
351 return FertilizerQuality >= FertilizerQuality.Deluxe ? 0f : Math.Max(0f, 1f - (this.getCropIridiumQualityChance() + this.getCropGoldQualityChance() + this.getCropSilverQualityChance()));
352 }

354 private double getCropSilverQualityChance()
355 {
356 return FertilizerQuality >= FertilizerQuality.Deluxe ? 1f - (this.getCropIridiumQualityChance() + this.getCropGoldQualityChance()) : (1f - this.getCropIridiumQualityChance()) * (1f - this.getCropBaseGoldQualityChance()) * Math.Min(0.75, 2 * this.getCropBaseGoldQualityChance());
357 }
