249 private double TotalCropProfit(Crop crop)
250 {
251 double totalProfitFromFirstProduce;
252 double totalProfitFromRemainingProduce;
253
254 if (!crop.affectByQuality)
255 {
256 totalProfitFromFirstProduce = 0;
257 totalProfitFromRemainingProduce = crop.Price;
258 }
259 else
260 {
261 double averageValue = crop.Price * this.GetAverageValueForCropAfterModifiers();//only applies to first produce
262 totalProfitFromFirstProduce = averageValue;
263
264 double averageExtraCrops = crop.AverageExtraCropsFromRandomness();
265
266 totalProfitFromRemainingProduce = (crop.MinHarvests - 1 >= 0 ? crop.MinHarvests - 1 : 0) * crop.Price;
267
268 totalProfitFromRemainingProduce += (crop.Price * averageExtraCrops);
269 }
270 if (!UseBaseStats && Game1.player.professions.Contains(Farmer.tiller))
271 {
272 totalProfitFromRemainingProduce *= 1.1f;
273 }
274 double result = (totalProfitFromFirstProduce + totalProfitFromRemainingProduce) * crop.TotalHarvestsWithRemainingDays(Season, FertilizerQuality, (int)Day);
275 return result;
276 }

436 {

444 //calculate average value modifier for price

472 private double GetCropBaseGoldQualityChance(double limit = 9999999999)
473 {
474 int fertilizerQualityLevel = ((int)FertilizerQuality) > 0 ? ((int)FertilizerQuality) : 0;
475 double part1 = (0.2 * (FarmingLevel / 10.0)) + 0.01;
476 double part2 = 0.2 * (fertilizerQualityLevel * ((FarmingLevel + 2) / 12.0));
477 return Math.Min(limit, part1 + part2);

484 private double GetCropBaseQualityChance()
485 {

493 private double GetCropSilverQualityChance()
494 {
495 return FertilizerQuality >= FertilizerQuality.Deluxe ? 1f - (this.GetCropIridiumQualityChance() + this.GetCropGoldQualityChance()) : (1f - this.GetCropIridiumQualityChance()) * (1f - this.GetCropBaseGoldQualityChance()) * Math.Min(0.75, 2 * this.GetCropBaseGoldQualityChance());
496 }
